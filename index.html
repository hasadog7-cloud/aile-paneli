Akıllı Performans Analiz Motoru ("Hastalıklı Kişi" Tespiti) function analizEtAileSagligi(tumGorevler) { let uyeIstatistikleri = {}; // Tüm görev geçmişini tara Object.values(tumGorevler).forEach(gorev => { if (!uyeIstatistikleri[gorev.assignedTo]) { uyeIstatistikleri[gorev.assignedTo] = { toplam: 0, patlayan: 0 }; } uyeIstatistikleri[gorev.assignedTo].toplam++; if (gorev.status === "zaman-asimi") { uyeIstatistikleri[gorev.assignedTo].patlayan++; } }); let riskliUyeler = {}; Object.keys(uyeIstatistikleri).forEach(uye => { const stats = uyeIstatistikleri[uye]; const failRate = stats.toplam > 0 ? (stats.patlayan / stats.toplam) : 0; // Görevlerin %30'undan fazlasını patlatan üye riskli (hastalıklı) kabul edilir riskliUyeler[uye] = failRate >= 0.30; }); return riskliUyeler; // Örn: { "baba": false, "alexa": false, "X_Uye": true } } // 2.
